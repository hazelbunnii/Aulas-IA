SENSOR DE LUZ NOTURNA

fun main() {
    //Percepção (Ambiente: "escuro" ou "claro")
    val ambiente = "escuro"
    
    //Regra Racional (Tomada de decisão)
    if (ambiente == "escuro") {
        println("[Atuador]: Lâmpada ACESA")
    }else{
        println("[Atuador]: Lâmpada APAGADA")
    }
}


O AR CONDICIONADO/TERMOSTATO

fun main() {
   val temperaturaAlvo = 22
   val temperaturaAtual = 25
